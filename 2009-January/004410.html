C# treats a monitor (lock statement) the same as Java. The same thread can
reenter a lock as many times as it wishes. The following code completes as
expected on .NET 3.5 SP1, Mono 1.9.1 (Windows), and Mono 1.9.1 (Ubuntu 8.10
AMD64):

using System;

namespace sandbox
{
    class Program
    {
        static void Main(string[] args)
        {
            object a = new object();

            Console.WriteLine("Starting");

            lock (a)
            {
                Console.WriteLine("Inside Main() lock");

                lock (a)
                {
                    Console.WriteLine("Inside second Main() lock");

                    F(a);
                }
            }

            Console.WriteLine("Done");
        }

        static void F(object a)
        {
            Console.WriteLine("Inside F()");

            lock (a)
            {
                Console.WriteLine("Inside F() lock");
            }
        }
    }
}
