following) and then run the opensim.exe from bin directory.

[A]
RegionUUID = 7d3b4466-a95e-4f51-8bb0-0103a6c1c8c1
Location = 1000,1000
InternalAddress = 0.0.0.0
InternalPort = 9000
AllowAlternatePorts = False
ExternalHostName = SYSTEMIP
